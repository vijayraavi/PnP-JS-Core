/**
 * Describes a collection of Item objects
 *
 */
class Items extends queryable_1.QueryableCollection {
    /**
     * Creates a new instance of the Items class
     *
     * @param baseUrl The url or Queryable which forms the parent of this fields collection
     */
    constructor(baseUrl, path = "items") {
        super(baseUrl, path);
    }
    /**
     * Gets an Item by id
     *
     * @param id The integer id of the item to retrieve
     */
    getById(id) {
        let i = new Item(this);
        i.concat(`(${id})`);
        return i;
    }
    /**
     * Skips the specified number of items (https://msdn.microsoft.com/en-us/library/office/fp142385.aspx#sectionSection6)
     *
     * @param skip The starting id where the page should start, use with top to specify pages
     */
    skip(skip) {
        this._query.add("$skiptoken", encodeURIComponent(`Paged=TRUE&p_ID=${skip}`));
        return this;
    }
    /**
     * Gets a collection designed to aid in paging through data
     *
     */
    getPaged() {
        return this.getAs(new PagedItemCollectionParser());
    }
    /**
     * Adds a new item to the collection
     *
     * @param properties The new items's properties
     */
    add(properties = {}, listItemEntityTypeFullName = null) {
        let doAdd = (listItemEntityType) => {
            let postBody = JSON.stringify(util_1.Util.extend({
                "__metadata": { "type": listItemEntityType },
            }, properties));
            return this.postAs({ body: postBody }).then((data) => {
                return {
                    data: data,
                    item: this.getById(data.Id),
                };
            });
        };
        if (!listItemEntityTypeFullName) {
            let parentList = this.getParent(lists_1.List);
            let removeDependency = this.addBatchDependency();
            return parentList.getListItemEntityTypeFullName().then(n => {
                let promise = doAdd(n);
                removeDependency();
                return promise;
            });
        }
        else {
            return doAdd(listItemEntityTypeFullName);
        }
    }
}

/**
 * Provides paging functionality for list items
 */
class PagedItemCollection {
    constructor(nextUrl, results) {
        this.nextUrl = nextUrl;
        this.results = results;
    }
    /**
     * If true there are more results available in the set, otherwise there are not
     */
    get hasNext() {
        return typeof this.nextUrl === "string" && this.nextUrl.length > 0;
    }
    /**
     * Gets the next set of results, or resolves to null if no results are available
     */
    getNext() {
        if (this.hasNext) {
            let items = new Items(this.nextUrl, null);
            return items.getPaged();
        }
        return new Promise(r => r(null));
    }
}
exports.PagedItemCollection = PagedItemCollection;
class PagedItemCollectionParser extends odata_1.ODataParserBase {
    parse(r) {
        return new Promise((resolve, reject) => {
            if (this.handleError(r, reject)) {
                r.json().then(json => {
                    let nextUrl = json.hasOwnProperty("d") && json.d.hasOwnProperty("__next") ? json.d.__next : json["odata.nextLink"];
                    resolve(new PagedItemCollection(nextUrl, this.parseODataJSON(json)));
                });
            }
        });
    }
}
class ItemUpdatedParser extends odata_1.ODataParserBase {
    parse(r) {
        return new Promise((resolve, reject) => {
            if (this.handleError(r, reject)) {
                resolve({
                    "odata.etag": r.headers.get("etag"),
                });
            }
        });
    }
}
